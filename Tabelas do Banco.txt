CREATE TABLE tb_cliente(
    CPF VARCHAR(14) PRIMARY KEY NOT NULL,
    NOME VARCHAR(30) NOT NULL,
    EMAIL VARCHAR(30) NOT NULL,
    HABILITACAO VARCHAR(12) NOT NULL,
    PLACA VARCHAR(10) NOT NULL,
    NASCIMENTO DATE NOT NULL,
    TELEFONE VARCHAR(14) NOT NULL,
    TIPO_DE_CLIENTE(30) NOT NULL,
    CONSTRAINT FOREIGN KEY(PLACA)
    REFERNCES tb_veiculo(NUMERO_DA_PLACA)

);

CREATE TABLE tb_veiculo(
    NUMERO_DA_PLACA VARCHAR(10) PRIMARY KEY NOT NULL,
    MARCA VARCHAR(30) NOT NULL,
    MODELO VARCHAR(30) NOT NULL,
    TIPO VEICULO VARCHAR(15) NOT NULL,
    
);


CREATE TABLE tb_usuario(
    USUARIO VARCHAR(10) PRIMARY KEY NOT NULL,
    SENHA VARCHAR(10) NOT NULL,
    FUNCAO VARCHAR(30) NOT NULL,
    
    
);
